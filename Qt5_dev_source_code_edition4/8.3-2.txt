#include <QAbstractItemView>
#include <QItemSelectionModel>
#include <QRegion>
#include <QMouseEvent>

{
    Q_OBJECT
public:
    HistogramView(QWidget *parent=0);
    //虚函数声明											//(a)
    QRect visualRect(const QModelIndex &index)const;

    //为selections赋初值
