            }
            if(hobby.children[7].checked) {	//若“其他”复选框选中
                hobbyText += "...";
            }
		    /* 最终生成的完整学生信息 */
            details.text = "我的名字叫" + name.text + "，是个" + age.text + "岁" +
                    sex.current.text + "生，所学专业是" + spec.currentText +

        }
    }
}
